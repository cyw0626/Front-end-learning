var arr=[{'a':1},{'a':3},{'a':2}];
function quickSort(arr){
    var middle=Math.floor(arr.length/2),
        base=arr.splice(middle,1);
        left=[],
        right=[];
    for(var i=0;i<arr.length;i++){
        if(arr[i]['a']<base['a']){
            left.push(arr[i]['a']);
        }else{
            right.push(arr[i]['a']);
        }
    }
    return quickSort(left).concat(base,quickSort(right));
}
quickSort(arr);
